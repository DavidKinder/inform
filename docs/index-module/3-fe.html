    for (int row = 0; row < 6; ) {
        int col = 0;
        unsigned int B, line_ended = FALSE;
        while (BinaryFiles::read_int8(INTERNAL_FILE, &B)) {
            if ((B == 10) || (B == 13)) { line_ended = TRUE; break; }
            if (col++ >= 80) { WRITE_TO(preview, "[...]"); line_ended = TRUE; break; }
